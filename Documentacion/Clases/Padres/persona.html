  7 import java.util.Objects;
  8 
  9 /**
 10  *
 11  * @author alumno
 12  */
 13 public abstract class Persona {
 14 
 15     private String nombre;
 16     private String DNI;
 17     private String apellidos;
 18     private int edad;
 19 
 20     public Persona(String nombre, String DNI, String apellidos, int edad) {
 21         this.nombre = nombre;
 22         this.DNI = DNI;
 23         this.apellidos = apellidos;
 24         this.edad = edad;
 25     }
 26 
 27     public Persona(Persona p) {
 28         this.nombre = p.getNombre();
 29         this.DNI = p.getDNI();
 30         this.apellidos = p.getApellidos();
 31         this.edad = p.getEdad();
 32     }
 33 
 34     public Persona() {
 35         this.nombre = "";
 36         this.DNI = "";
 37         this.apellidos = "";
 38         this.edad = 0;
 39     }
 40 
 41     @Override
 42     public String toString() {
 43         return "Nombre: " + nombre + ", apellidos: " + apellidos + ", DNI: " + DNI + ", edad: " + edad;
 44     }
 45 
 46     public String getNombre() {
 47         return nombre;
 48     }
 49 
 50     public void setNombre(String nombre) {
 51         this.nombre = nombre;
 52     }
 53 
 54     public String getDNI() {
 55         return DNI;
 56     }
 57 
 58     public void setDNI(String DNI) {
 59         this.DNI = DNI;
 60     }
 61 
 62     public String getApellidos() {
 63         return apellidos;
 64     }
 65 
 66     public void setApellidos(String apellidos) {
 67         this.apellidos = apellidos;
 68     }
 69 
 70     public int getEdad() {
 71         return edad;
 72     }
 73 
 74     public void setEdad(int edad) {
 75         this.edad = edad;
 76     }
 77 
 78     @Override
 79     public int hashCode() {
 80         int hash = 5;
 81         hash = 67 * hash + Objects.hashCode(this.nombre);
 82         hash = 67 * hash + Objects.hashCode(this.DNI);
 83         hash = 67 * hash + Objects.hashCode(this.apellidos);
 84         hash = 67 * hash + this.edad;
 85         return hash;
 86     }
 87 
 88     @Override
 89     public boolean equals(Object obj) {
 90         if (this == obj) {
 91             return true;
 92         }
 93         if (obj == null) {
 94             return false;
 95         }
 96         if (getClass() != obj.getClass()) {
 97             return false;
 98         }
 99         final Persona other = (Persona) obj;
100         if (this.edad != other.edad) {
101             return false;
102         }
103         if (!Objects.equals(this.nombre, other.nombre)) {
104             return false;
105         }
106         if (!Objects.equals(this.DNI, other.DNI)) {
107             return false;
108         }
109         return Objects.equals(this.apellidos, other.apellidos);
110     }
111 
112 }
