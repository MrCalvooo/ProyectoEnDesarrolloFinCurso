 13 public class Gerente extends Empleado {

 19     @Override

 22     }
 23 
 24     /*Constructores*/
 25     //Completo
 26     public Gerente(Jefe j, double complemento, int idGerente, String nombre, String DNI, String apellidos, int edad, double salario) {
 27         super(nombre, DNI, apellidos, edad, salario);
 28         this.j = j;
 29         this.complemento = complemento;
 30         this.idGerente = idGerente;
 31     }
 32 
 33     //Copia
 34     public Gerente(Jefe j, Gerente g, Empleado e, Persona p) {
 35         super(e, p);
 36         this.j = j;
 37         this.complemento = g.getComplemento();
 38         this.idGerente = g.getIdGerente();
 39     }
 40 
 41     //Vacio
 42     public Gerente() {
 43         super();
 44         this.complemento = 0.;
 45     }
 46

 63     }
 64 
 65     @Override
 66     public int hashCode() {
 67         int hash = 5;
 68         hash = 79 * hash + Objects.hashCode(this.j);
 69         hash = 79 * hash + (int) (Double.doubleToLongBits(this.complemento) ^ (Double.doubleToLongBits(this.complemento) >>> 32));
 70         return hash;
 71     }
 72 
 73     @Override
 74     public boolean equals(Object obj) {
 75         if (this == obj) {
 76             return true;
 77         }
 78         if (obj == null) {
 79             return false;
 80         }
 81         if (getClass() != obj.getClass()) {
 82             return false;
 83         }
 84         final Gerente other = (Gerente) obj;
 85         if (Double.doubleToLongBits(this.complemento) != Double.doubleToLongBits(other.complemento)) {
 86             return false;
 87         }
 88         return Objects.equals(this.j, other.j);
 89     }

 91     public int getIdGerente() {

 93     }

 96         this.idGerente = idGerente;
