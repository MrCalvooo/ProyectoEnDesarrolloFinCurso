 13 public class Coche extends Vehiculo {
 14 
 15     private int puertas;
 16     private String modelo;
 17     private int luces;
 18     private double capacidadMaletero;
 19 
 20     @Override
 21     public String toString() {
 22         return "Koenigsegg: " + modelo + " " + super.toString() + ",\npuertas: " + puertas + ", luces: " + luces + ", capacidadMaletero: " + capacidadMaletero;
 23     }
 24 
 25     public Coche(int puertas, String modelo, int luces, double capacidadMaletero, int id, String color, String motor, int ruedas, int asientos, double alto, double ancho, String tipoCombustible, double capacidadDeposito, double velocidadMaxima, double precio) {
 26         super(id, color, motor, ruedas, asientos, alto, ancho, tipoCombustible, capacidadDeposito, velocidadMaxima, precio);
 27         this.puertas = puertas;
 28         this.modelo = modelo;
 29         this.luces = luces;
 30         this.capacidadMaletero = capacidadMaletero;
 31     }
 32 
 33     public Coche() {
 34         super();
 35         this.puertas = 0;
 36         this.modelo = "";
 37         this.luces = 0;
 38         this.capacidadMaletero = 0.;
 39     }
 40 
 41     public Coche(Coche c, Vehiculo v) {
 42         super(v);
 43         this.puertas = c.getPuertas();
 44         this.modelo = c.getModelo();
 45         this.luces = c.getLuces();
 46         this.capacidadMaletero = c.getCapacidadMaletero();
 47     }

 77 
 78     public void setCapacidadMaletero(double capacidadMaletero) {

 80     }
 81 
 82     //equals y hashCode
 83     @Override
 84     public int hashCode() {
 85         int hash = 3;
 86         hash = 79 * hash + this.puertas;
 87         hash = 79 * hash + Objects.hashCode(this.modelo);
 88         hash = 79 * hash + this.luces;
 89         hash = 79 * hash + (int) (Double.doubleToLongBits(this.capacidadMaletero) ^ (Double.doubleToLongBits(this.capacidadMaletero) >>> 32));
 90         return hash;
 91     }
 92 
 93     @Override
 94     public boolean equals(Object obj) {
 95         if (this == obj) {
 96             return true;
 97         }
 98         if (obj == null) {
 99             return false;
100         }
101         if (getClass() != obj.getClass()) {
102             return false;
103         }
104         final Coche other = (Coche) obj;
105         if (this.puertas != other.puertas) {
106             return false;
107         }
108         if (this.luces != other.luces) {
109             return false;
110         }
111         if (Double.doubleToLongBits(this.capacidadMaletero) != Double.doubleToLongBits(other.capacidadMaletero)) {
112             return false;
113         }
114         return Objects.equals(this.modelo, other.modelo);
115     }
